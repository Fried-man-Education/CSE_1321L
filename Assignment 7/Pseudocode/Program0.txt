            CASE 1: letter ← 'C'
                BREAK
            CASE 2: letter ← 'A'
                BREAK
            CASE 3: letter ← 'T'
        END SWITCH
        PRINT letter
    END FOREACH

END MAIN

METHOD ConvertAndPrint (parameters: input)
BEGIN
    FOR i ← 0 to input.Length
        IF (input[i] == 'T') THEN
            input[i] ← 'U'
        ENDIF
        PRINT input[i]
    END FOR
END ConvertAndPrint
